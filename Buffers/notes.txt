//1. Methods to create Buffer
1 var buf=new Buffer(10);
2 var buf=new Buffer([10,20,30,40]);
3 var buf=new Buffer("Simply Easy");


//2. Writing to Buffers
buf.write(string[, offset][, length][, encoding])

string:data to be written to buffer
offset:index of buffer to start writing at
length:number of bytes to write
encoding:to use Encoding


//3. Reading from Buffers
buf.toString([encoding][, start][, end])

encoding:to use encoding
start:Beginning index to start reading
end:End index to end reading(Default is comlete buffer)


//4. Convert buffer to JSON
buf.toJSON()


//5. Concateanate Buffer
Buffer.concat(list[, totalLength])
list: Array List of Buffer Objects to be concatenated
totalLength: total Length of buffers when concatenated


//6. Compare Buffers //Returns a number indicating whether it comes before or after or is the same as the otherBuffer in sort order.
buf.compare(otherBuffer);


//7. Copy Buffers 
buf.copy(targetBuffer[, targetStart][, sourceStart][, sourceEnd])
targetBuffer − Buffer object where buffer will be copied.
targetStart − Number, Optional, Default: 0
sourceStart − Number, Optional, Default: 0
sourceEnd − Number, Optional, Default: buffer.length


//8. Slice Buffer //Returns a new buffer which references the same memory as the old one, but offset and cropped by the start (defaults to 0) and end (defaults to buffer.length) indexes. Negative indexes start from the end of the buffer.
buf.slice([start][, end])
start: Number ; end:Number

buf.length
